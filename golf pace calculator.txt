<script>
function calculateHole() {
    const teeTimeInput = document.getElementById('teeTime').value;
    const minsPerHole = parseFloat(document.getElementById('pace').value);
    
    if (!teeTimeInput) {
        alert("Please enter a tee time");
        return;
    }

    const now = new Date();
    const [hours, minutes] = teeTimeInput.split(':');
    const teeTime = new Date();
    teeTime.setHours(hours, minutes, 0);

    // Calculate difference in minutes
    let diffMs = now - teeTime;
    let elapsedMins = Math.floor(diffMs / 60000);

    if (elapsedMins < 0) {
        document.getElementById('result').innerHTML = "You haven't started yet!";
        document.getElementById('elapsed').innerHTML = "";
        return;
    }

    // Calculate current hole
    let hole = Math.floor(elapsedMins / minsPerHole) + 1;
    
    if (hole > 18) {
        document.getElementById('result').innerHTML = "Round Finished! ??";
        document.getElementById('elapsed').innerHTML = `Total time: ${elapsedMins} mins`;
    } else {
        document.getElementById('result').innerHTML = `You should be on Hole #${hole}`;
        document.getElementById('elapsed').innerHTML = `Time elapsed: ${elapsedMins} minutes`;
    }
}
</script>
